	\Battr{nombre}{Nombre}{\tdFrase}{Nombre de la persona}{\requerido}{\longitudMinMax{2}{caracteres}{20}{caracteres}}
	\Battr{primerApellido}{Primer Apellido}{\tdFrase}{Primer apellido de la persona}{\requerido}{\longitudMinMax{2}{caracteres}{20}{caracteres}}
	\Battr{segundoApellido}{Segundo Apellido}{\tdFrase}{Segundo apellido de la persona}{\opcional}{\longitudMinMax{2}{caracteres}{20}{caracteres}}

	\Battr{contrasenia}{Contraseña}{\tdFrase}{Contraseña de la cuenta}{\requerido}{\longitudMinMax{6}{caracteres}{20}{caracteres}}
\end{BusinessEntity}

\section{Modelo de información: Trabajo Terminal}

En esta sección se describen las entidades del módulo Trabajo Terminal.

%--------------------------------------------------------------------------------
\begin{BusinessEntity}{Protocolo}{Protocolo}
	\Battr{numeroProtocolo}{Número del protocolo}{\tdFrase}{Número del protocolo que le fue asignado por los Gestores TT o por el Secretario Ejecutivo}\opcional
	\Battr{titulo}{Título}{\tdFrase}{Título del protocolo}{\requerido}{\longitudMax{255}{caracteres}}
	\Battr{palabrasClave}{Palabras Clave}{\tdFrase}{Palabras claves que tienen relación con el protocolo}{\requerido}
	\Battr{resumen}{Resumen}{\tdFrase}{Breve descripción del protocolo}{\requerido}{\longitudMax{5000}{caracteres}}
	\Battr{objetivo}{Objetivo}{\tdFrase}{Resultado final del trabajo a entregar en el Trabajo Terminal II}{\requerido}{\longitudMax{5000}{caracteres}}
	\Battr{documentoProtocolo}{Documento del Protocolo}{\tdArchivo}{Es el documento en PDF del protocolo}{\requerido}{\longitudMax{5}{MB}}
	\Battr{observaciones}{Observaciones}{\tdFrase}{Es un comentario donde se anotan las observaciones que tiene el protocolo}{\requerido}{\longitudMax{1000}{caracteres}}
	\Battr{periodo}{Periodo}{\tdFrase}{Es el periodo en que se registró el protocolo}\opcional
	\Battr{estado}{Estado}{\tdFrase}{Es el estado en el que se encuentra el protocolo. Depende de la fase del proceso en la que esté}\requerido
\end{BusinessEntity}
