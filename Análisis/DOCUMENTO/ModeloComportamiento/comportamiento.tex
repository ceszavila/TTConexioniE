\section{Actores del sistema}\label{sec:Comportamiento:ActoresSistema}

Los actores son aquellos usuarios que van a interactuar con el sistema, los actores para el SREP se describen a continuación.

%----------------------------------------------Alumno-----------------------------------------------
\begin{actor}{Alumno}{Persona encargada de registrar el \cdtRef{gls:protocolo}{protocolo} en el sistema.} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Registrar el \cdtRef{gls:protocolo}{protocolo} en el sistema y sus correspondientes datos.
		\item Invitar a otros alumnos a formar parte del \cdtRef{gls:protocolo}{protocolo}.
		\item Registrar a los directores que formarán parte del \cdtRef{gls:protocolo}{protocolo}.
		\item Realizar las modificaciones necesarias cuando el \cdtRef{gls:protocolo}{protocolo} sea mandado a reestructuración.
	\end{itemize}
\end{actor}


%---------------------------------Director Externo------------------------------------------------
\begin{actor}{Director Externo}{\cdtRef{actor:Persona}{Persona} ajena a la ESCOM encargada de dirigir un Trabajo Terminal.} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Aceptar o rechazar participar en un \cdtRef{gls:protocolo}{protocolo}.
		\item Verificar en cuantos \cdtRef{gls:protocolo}{protocolos} tiene participación.
		\item Verificar el \cdtRef{gls:estadoProtocolo}{estado de los protocolos} que dirige.
	\end{itemize}
\end{actor}


%---------------------------------Secretario Ejecutivo------------------------------------------------
\begin{actor}{Secretario Ejecutivo}{\cdtRef{actor:Profesor}{Profesor} de la ESCOM encargado de coordinar a los \cdtRef{actor:Gestor TT}{Gestor TT}} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Administrar a los \cdtRef{actor:Director Externo}{directores externos}.
		\item Modificar correos electrónicos de los usuarios registrados.
		\item Administrar el calendario de actividades.
		\item Carga a los \cdtRef{actor:Alumno}{alumnos} en el sistema.
		\item Carga a los \cdtRef{actor:Gestor TT}{Gestores TT} en el sistema.
	\end{itemize}
\end{actor}


%---------------------------------Presidente de academia------------------------------------------------
\begin{actor}{Presidente de Academia}{\cdtRef{actor:Profesor}{Profesor} que dirige una academia de la ESCOM} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item  Asignar Sinodales a un \cdtRef{gls:protocolo}{protocolo}.
	\end{itemize}
\end{actor}

\end{actor}

%---------------------------------Gestor TT------------------------------------------------
\begin{actor}{Gestor TT}{Personal administrativo que trabaja en la CATT de la ESCOM} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Modificar correos electrónicos de los usuarios registrados.
	\end{itemize}
	\item[Perfil:]

		\item Saber cómo usar una computadora.
