\section{Actores del sistema}\label{sec:Comportamiento:ActoresSistema}

Los actores son aquellos usuarios que van a interactuar con el sistema, los actores para el SREP se describen a continuación.

%----------------------------------------------Alumno-----------------------------------------------
\begin{actor}{Alumno}{Persona encargada de registrar el \cdtRef{gls:protocolo}{protocolo} en el sistema.} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Registrar el \cdtRef{gls:protocolo}{protocolo} en el sistema y sus correspondientes datos.
		\item Invitar a otros alumnos a formar parte del \cdtRef{gls:protocolo}{protocolo}.
		\item Registrar a los \cdtRef{actor:Director}{Director} que formarán parte del \cdtRef{gls:protocolo}{protocolo}.
		\item Realizar las modificaciones necesarias cuando el \cdtRef{gls:protocolo}{protocolo} fue mandado a reestructuración.
	\end{itemize}
\end{actor}


%-----------------------------------------Director------------------------------------------------
\begin{actor}{Director}{\cdtRef{actor:Profesor}{Profesor} de la ESCOM encargado de asesorar a los \cdtRef{actor:Alumno}{alumnos} en la creación del \cdtRef{gls:protocolo}{protocolo}} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Aceptar o rechazar participar en un \cdtRef{gls:protocolo}{protocolo}.
		\item Verificar en cuantos \cdtRef{gls:protocolo}{protocolos} tiene participación.
		\item Verificar el \cdtRef{gls:estadoProtocolo}{estado de los protocolos} que dirige.
	\end{itemize}
\end{actor}


%---------------------------------Director Externo------------------------------------------------
\begin{actor}{Director Externo}{\cdtRef{actor:Profesor}{Profesor} externo a la ESCOM encargado de asesorar a los \cdtRef{actor:Alumno}{alumnos} en la creacion del \cdtRef{gls:protocolo}{protocolo}} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Ingresar sus datos personales y requeridos por la CATT en el sistema.
		\item Verificar el \cdtRef{gls:estadoProtocolo}{estado de los protocolos} que dirige.
	\end{itemize}
\end{actor}


%---------------------------------Secretario Ejecutivo------------------------------------------------
\begin{actor}{Jefe CATT}{\cdtRef{actor:Profesor}{Profesor} de la ESCOM encargado de coordinar a los \cdtRef{actor:Administrador CATT}{Trabajadores de la CATT}} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item Carga a los \cdtRef{actor:Alumno}{alumnos} en el sistema.
		\item Carga a los \cdtRef{actor:Administrador CATT}{Trabajadores de la CATT} en el sistema.
	\end{itemize}
\end{actor}


%---------------------------------Presidente de academia------------------------------------------------
\begin{actor}{Presidente de Academia}{\cdtRef{actor:Profesor}{Profesor} que dirige una academia de la ESCOM} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item  Asigna las sinodalias de un \cdtRef{gls:protocolo}{protocolo}.
	\end{itemize}
\end{actor}


%---------------------------------Profesor------------------------------------------------
\begin{actor}{Profesor}{Docente que imparte cátedra en la ESCOM} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item  Evalúa los \cdtRef{gls:protocolo}{protocolos} de los que es sinodal.
		\item  Manda a restructuración los \cdtRef{gls:protocolo}{protocolos} de los que es sinodal.
		\item  Declina una sinodalia.
	\end{itemize}
\end{actor}

%---------------------------------Administrador CATT------------------------------------------------
\begin{actor}{Administrador CATT}{Personal administrativo que trabaja en la CATT de la ESCOM} 
	\item[Responsabilidades:] \hspace{1pt}
	\begin{itemize}
		\item  Modifica contraseñas de \cdtRef{actor:Alumno}{alumnos} o \cdtRef{actor:Profesor}{profesores}
		\item  Avala finalización de inscripción de protocolo.
	\end{itemize}
\end{actor}
