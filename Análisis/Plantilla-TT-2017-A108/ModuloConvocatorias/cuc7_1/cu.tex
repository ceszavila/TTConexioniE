\begin{UseCase}{CUAC-7.1}{Registrar convocatoria de cursos}{
		Permite llevar a cabo el registro de una nueva convocatoria de cursos para la Escuela Superior de Cómputo debido a que será impartido un nuevo curso en dicha escuela. \\
		}
		\UCitem{Versión}{1.0}
		\UCccsection{Administración de Requerimientos}
		\UCitem{Autor}{Ivo Sebastián Sam Álvarez-Tostado}
		\UCccitem{Evaluador}{Ulises Velez Saldaña}
		\UCitem{Operación}{Consulta}
		\UCccitem{Prioridad}{Alta}
		\UCccitem{Complejidad}{Baja}
		\UCccitem{Volatilidad}{Baja}
		\UCccitem{Madurez}{Alta}
		\UCitem{Estatus}{Por revisar}
		\UCitem{Fecha del último estatus}{24 de Mayo del 2018}
			
		\UCsection{Atributos}
		\UCitem{Actor}{
			\begin{UClist} 
				\UCli \cdtRef{actor:CIEUtec}{Responable UTEC}
			\end{UClist}
		}
		\UCitem{Propósito}{Proporcionar un medio por el cual se pueda registrar nuevas convocatorias de cursos para la Escuela Superior de Cómputo.}
		\UCitem{Entradas}{
			\begin{UClist}
				\UCli Nombre del curso
				\UCli Fecha de inicio del curso
				\UCli Fecha de término del curso
				\UCli Costo del curso para externos
				\UCli Costo del curso para estudiantes del IPN
				\UCli Descripción del curso
				\UCli Requisitos del curso
				\UCli URL del curso
			\end{UClist}
		}
		\UCitem{Salidas}{
			\begin{UClist}
				\UCli Nombre del curso
				\UCli Fecha de inicio del curso
				\UCli Fecha de término del curso
				\UCli Costo del curso para externos
				\UCli Costo del curso para estudiantes del IPN
				\UCli Descripción del curso
				\UCli Requisitos del curso
				\UCli URL del curso
			\end{UClist}
		}
		\UCitem{Precondiciones}{\textbf{Externa}: Se debió aprobar la convocatoria para un nuevo curso.}
		\UCitem{Postcondiciones}{Se podrá consultar la convocatoria del curso}
		\UCitem{Reglas de negocio}{\cdtIdRef{RN-S1}{Datos Obligatorios}}
		\UCitem{Errores}{Ninguno}
		\UCitem{Tipo}{Primario.}
	\end{UseCase}
	
	\begin{UCtrayectoria}
		
		\UCpaso [\UCactor] Solicta registrar una nueva convocatoria de curso tocando el botón \textbf{Registrar convocatoria de curso} de la pantalla \textbf{Gestionar convocatorias de cursos}.
		
		\UCpaso Muestra la pantalla \textbf{Registrar convocatoria de curso}.
		
		\UCpaso [\UCactor] Ingresa los datos solicitados.
		
		\UCpaso [\UCactor] Solicita finalizar la operación tocando el botón \textbf{Registrar}. \refTray{A}
		
		\UCpaso Verifica que se haya ingresado los campos marcados como obligatorios, con base en la regla \cdtIdRef{RN-S1}{Datos Obligatorios}.
		
		\UCpaso Persiste la información registrada.
		
		\UCpaso Regresa a la pantalla \textbf{Gestionar convocatorias de cursos}.
		
	\end{UCtrayectoria}

	\begin{UCtrayectoriaA}{A}{Cuando el actor requiere cancelar la operación.}
		
		\UCpaso [\UCactor] Solicita cancelar la operación tocando el botón \textbf{Volver}.
		
		\UCpaso Regresa a la pantalla.
		
	\end{UCtrayectoriaA}
