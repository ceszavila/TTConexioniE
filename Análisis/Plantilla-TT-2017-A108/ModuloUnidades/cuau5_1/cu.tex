	\begin{UCtrayectoria}
		\UCpaso[\UCactor] Presiona el botón \cdtButton{Registrar UA} de la pantalla \cdtIdRef{IUAU-5}{Gestionar Unidades de Aprendizaje}
		\UCpaso[\UCsist] Construye la pantalla \cdtIdRef{IUAU-5.1}{Registrar Unidad de Aprendizaje}.
		\UCpaso[\UCsist] Muestra en la pantalla \cdtIdRef{IUAU-5.1}{Registrar Unidad de Aprendizaje} con el botón \cdtButton{Registrar Unidad de Aprendizaje}. \label{CUU5.1:uaRegistrar}.
		\UCpaso[\UCactor] Ingresa los datos solicitados por la pantalla \cdtIdRef{IUAU-5.1}{Registrar Unidad de Aprendizaje}.\label{CUAU5.1Paso}
		\UCpaso[\UCactor] Presiona el botón \cdtButton{Registrar Unidad de Aprendizaje} de la pantalla \cdtIdRef{IUAU-5.1}{Registrar Unidad de Aprendizaje}.
		\UCpaso[\UCsist] Verifica que no se omitan ninguno de los campos marcados como obligatorio como se indica en la regla de negocios \cdtIdRef{RN-S1}{Datos obligatorios}. \refTray{A}
		\UCpaso[\UCsist] Registra la unidad de aprendizaje en el sistema.
		\UCpaso[\UCsist] Construye el mensaje \cdtIdRef{MSG01}{Operación exitosa} con los valores \textit{VALOR = La unidad de aprendizaje y OPERACIÓN = registró}.
		\UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG01}{Operación exitosa} en la pantalla \cdtIdRef{IUAU-5.1}{Registrar Unidad}.
	 	\end{UCtrayectoria}
	
	\begin{UCtrayectoriaA}{A}{Faltan campos obligatorios.}
		\UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG02}{Datos Obligatorios} en la pantalla  \cdtIdRef{IUAU-5.1}{Registrar Unidad de Aprendizaje}.
		\UCpaso[] Continua en el paso \ref{CUAU5.1Paso} de la trayectoria principal.
	\end{UCtrayectoriaA}
